\vspace{0.2em}
\scriptsize
\begin{tabular}{lcc}
\toprule

\midrule

\bottomrule
\end{tabular}

\vspace{0.4em}
\begin{exampleblock}{\scriptsize Key Finding}
\scriptsize GPU 利用率始终 100\%，\textbf{视频解码是真正瓶颈}。torchcodec GPU 硬件加速 >> pyav CPU 逐帧解码。
\end{exampleblock}

\vspace{0.3em}
\begin{alertblock}{\scriptsize Why batch\_size=128 Max?}
\scriptsize ACT chunk\_size=100 $O(n^2)$，128 batch 时显存 ${\sim}$79GB（RTX 6000D 上限）。
\end{alertblock}
